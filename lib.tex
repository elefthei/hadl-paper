%%%%%%%%%%%%% Pact formal model — judgments
% RtType is now envless: v : τ. Kept as shorthand for the ok-branch of
% the type-check judgment \TJok on a closed value.
\newcommand{\RtType}[2]{#1 : #2}
\newcommand{\StType}[2]{\vdash #1 : #2}
% Typing-context metavariable for the schema-name environment.
\newcommand{\Gctx}{\Gamma}
% Type-check judgment — single-valued, returns either \mathsf{ok} (success)
% or \mathsf{err}(ε) where ε is the typing-error string fed back to the
% oracle during self-healing. Applies to both values (runtime recheck of
% an oracle response) and expressions (continuation check that drives
% self-heal per pact-formal.md). Matches pact-formal.md.
\newcommand{\TJok}[3]{#1 \vdash #2 : #3 \;\Rightarrow\; \mathsf{ok}}
\newcommand{\TJerr}[4]{#1 \vdash #2 : #3 \;\Rightarrow\; \mathsf{err}(#4)}
% Context extension \Gamma, x : \tau used in continuation-driven heal
% rules. Paired with \TJok / \TJerr to type-check the continuation p.
\newcommand{\Gext}[2]{\Gctx,\, #1 : #2}
\newcommand{\PolOk}[3]{#1;\,#2 \vdash #3 : \mathsf{ok}}
