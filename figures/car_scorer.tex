workflow pick_car() -> String {
  let prefs = ask "What matters most to you?"
  let car_schema: Schema = gen "Generate a
    schema for a car with a name and a type."
  let scorer: (car_schema -> Number) =
    gen "Create a car scoring workflow based on `prefs`."
  var best: String = ""
  var top: Number = 0
  var i: Number = 0
  while (i < 3) {
    let car: car_schema =
      gen "Come up with a popular car matching `prefs`."
    let score: Number = eval scorer(car)
    if (score > top) {
      top = score
      best = car.name
    }
    say "`car.name` rated `score`."
    i = i + 1
  }
  say "Recommended `best`."
  best
}
\end{lstlisting}
